import { Roompanda } from "@roompanda/sdk";

const panda = new Roompanda(process.env.PANDA_KEY);

const rooms = await panda.availability.list({
  propertyId: "the-laurel",
  checkin: "2026-09-12",
  checkout: "2026-09-15",
  guests: 2,
});

console.log(rooms[0].name, rooms[0].rate);
// => "Garden View King" 189
from roompanda import Roompanda

panda = Roompanda(api_key="sk_live_panda")

rooms = panda.availability.list(
    property_id="the-laurel",
    checkin="2026-09-12",
    checkout="2026-09-15",
    guests=2,
)

print(rooms[0].name, rooms[0].rate)
# => "Garden View King" 189
